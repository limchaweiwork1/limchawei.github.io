Preprocessing

  • Create pipelines for 2 reasons below:
  • imputation: replace the missing values
  • standardization: big range in numerical features
In [ ]:
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.neighbors import KNeighborsClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.tree import DecisionTreeClassifier

# Create preprocessing pipelines
mean_standardscaler_pipeline = Pipeline(steps=[
    ('imputer_mean', SimpleImputer(strategy='mean')), # replace missing data & feature's distribution is relatively symmetric and not heavily skewed by outliers.
    ('normalizer', StandardScaler())
])

median_standardscaler_pipeline = Pipeline(steps=[
    ('imputer_median', SimpleImputer(strategy='median')), # replace missing data & more robust to outliers
    ('normalizer', StandardScaler())
])

standardscaler_pipeline = Pipeline(steps=[
    ('normalizer', StandardScaler())
])

most_frequent_pipeline = Pipeline(steps=[
    ('imputer', SimpleImputer(strategy='most_frequent')) # replace missing data
])


# Use ColumnTransformer to apply preprocessing pipelines to the correct columns
preprocessor = ColumnTransformer(
    transformers=[
        ('num_mean_standardscaler_transform', mean_standardscaler_pipeline, ['Process T (C)']),
        ('num_median_standardscaler_transform', median_standardscaler_pipeline, ['Rotation Speed (rpm)']),
        ('num_standardscaler_transform', standardscaler_pipeline, ['Tool Wear (min)']),
        ('cat_most_frequent_transform', most_frequent_pipeline, ['Quality_L', 'Quality_M'])
    ],
    remainder='drop' # Any columns in dataset that are not explicitly mentioned in the transformers list will be dropped.
)

# Integrate into a full pipeline with K-Nearest Neighbors (KNN) model
# Using n_neighbors=6 as per your first snippet.
knn_full_pipeline = Pipeline(steps=[
    ('preprocessor', preprocessor),
    ('knn_classifier', KNeighborsClassifier())
])

# Integrate into a full pipeline with Logistic Regression model
lr_full_pipeline = Pipeline(steps=[
    ('preprocessor', preprocessor),
    ('logistic_regression', LogisticRegression())
])

# Integrate into a full pipeline with Decision Tree Classifier/Classification and Regression Trees (CART) model
dtc_full_pipeline = Pipeline(steps=[
    ('preprocessor', preprocessor),
    ('decision_tree_classifier', DecisionTreeClassifier())
])

  • Find the best parameter for the best model (Decision Tree Classifier)

  • Support:

    • Class 0: 3864 instances in the test set.
    • Class 1: 136 instances in the test set.
    • This shows a class imbalance due to Class 0 is majority.

  • For Class 0:

    • precision: 0.99: When the model predicts class 0, it is correct 99% of the time.
    • recall: 0.99: The model correctly identifies 99% of all actual class 0 instances.
    • f1-score: 0.99: A balanced measure of precision and recall, also excellent.

  • For Class 1:

    • precision: 0.67: When the model predicts class 1, it is correct only 67% of the time.
    • recall: 0.65: The model correctly identifies only 65% of all actual class 1 instances.
    • f1-score: 0.66: Reflects the poorer performance on class 1 compared to class 0.

  • 48 class 1 predicted as 0 and 43 class 0 predicted as class 1.

  • Reason: imbalanced data due to small minority class cause model cannot identify well.

from sklearn.model_selection import train_test_split, GridSearchCV
from sklearn.metrics import accuracy_score, classification_report, confusion_matrix

dtc_param_grid = {
    'decision_tree_classifier__criterion': ['gini', 'entropy'],
    'decision_tree_classifier__max_depth': [3, 5, 7, None],
    'decision_tree_classifier__min_samples_split': [2, 5, 10],
    'decision_tree_classifier__min_samples_leaf': [1, 2, 4],
    'decision_tree_classifier__max_features': ['sqrt', 'log2', None]
}

kf = KFold(n_splits=6, shuffle=True, random_state=42)

print('\nTuning Decision Tree Classifier:')
dtc_grid_search = GridSearchCV(
    estimator=dtc_full_pipeline,
    param_grid=dtc_param_grid,
    cv=kf
)
dtc_grid_search.fit(X_train, y_train)
print(f'\nBest parameters for Decision Tree Classifier: {dtc_grid_search.best_params_}')
print(f'Best cross-validation accuracy for Decision Tree Classifier: {dtc_grid_search.best_score_:.3f}') # performance measured on the training data with best combination of hyperparameters

# Evaluate the best tuned Decision Tree Classifier model on the test set
best_dtc_model = dtc_grid_search.best_estimator_
dtc_test_predictions = best_dtc_model.predict(X_test)
dtc_tuned_accuracy = accuracy_score(y_test, dtc_test_predictions)

print(f'\nTuned Decision Tree Classifier Test Accuracy: {dtc_tuned_accuracy:.3f}') # final performance measured on the independent testing data
print('Classification Report for Tuned Decision Tree Classifier on Test Set:')
print(classification_report(y_test, dtc_test_predictions))
print('Confusion matrix for Tuned Decision Tree Classifier on Test Set:')
print(confusion_matrix(y_test, dtc_test_predictions))
