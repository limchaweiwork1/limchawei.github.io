from keras.layers import Dense

# Design and implement a CNN model using the Sequential API in Keras
model = Sequential()
# Convolutional layers with appropriate filter sizes and activation functions (e.g., ReLU).
model.add(Conv2D(32, (5, 5), input_shape=(28,28,1), activation='relu'))  # (5, 5) is kernel_size
# Pooling layers (e.g., MaxPooling2D) for dimensionality reduction.
model.add(MaxPooling2D(pool_size=(2, 2)))
model.add(Conv2D(64, (3, 3), activation='relu'))
model.add(MaxPooling2D(pool_size=(2, 2)))
# Flattening layer to prepare data for fully-connected layers.
model.add(Flatten())
# Fully-connected layers with appropriate number of neurons for classification
model.add(Dense(128, activation='relu'))
model.add(Dense(number_classes, activation='softmax'))
